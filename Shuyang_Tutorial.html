import scipy.stats as stats
import statsmodels.api as sm
import numpy as np
import matplotlib.pyplot as plt


x = np.array(range(20))
y = 3 + 0.5 * x + 2 * np.random.randn(20)
#plot the data
plt.plot(x, y, 'bo')
plt.show()

results = sm.OLS(y, x).fit()
print(results.summary())

slope= results.params[0]
plt.plot(x, y, 'bo')
plt.hold(True)
# Plot a line
y =  slope * x
plt.plot(x, y, 'r-')
plt.show()
